let currentMatchIndex = -1; let matches = []; // Search function const performSearch = () => { const searchText = searchInput.value.trim().toLowerCase(); const contentElements = document.querySelectorAll("h1, h2, h3, p, span, div"); // Clear previous highlights contentElements.forEach(el => { el.innerHTML = el.innerHTML.replace(/(.*?)<\/mark>/g, "$1"); }); matches = []; currentMatchIndex = -1; if (searchText) { contentElements.forEach((el, index) => { const innerText = el.textContent.toLowerCase(); element.scrollIntoView({ behavior: "smooth", block: "center" }); } }; // Navigate to the previous match const navigatePrev = () => { if (matches.length > 0 && currentMatchIndex > 0) { currentMatchIndex--; navigateToMatch(); } }; // Navigate to the next match const navigateNext = () => { if (matches.length > 0 && currentMatchIndex < matches.length - 1) { currentMatchIndex++; navigateToMatch(); } }; // Event listeners searchButton.addEventListener("click", performSearch);
